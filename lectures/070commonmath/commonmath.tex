\sect{Monotonicity}
\begin{description}
  \item[monotonically increasing:] \[m \leq n \Rightarrow f(m) \leq f(n)\]
  \item[monotonically decreasing:] \[m \leq n \Rightarrow f(m) \geq f(n)\]
  \item[strictly increasing:] \[m < n \Rightarrow f(m) < f(n)\]
  \item[strictly decreasing:] \[m < n \Rightarrow f(m) > f(n)\]
\end{description}


\end{frame}

\sect{Floors and ceilings}
\begin{description}
  \item[floor:] $\floor{x}$ largest integer less than or equal to $x$
  \item[ceiling:] $\ceiling{x}$ smallest integer greater than or equal to $x$
\end{description}
\[
x-1 < \floor{x} \leq x \leq \ceiling{x} < x+1
\]
\begin{align*}
  \floor{n/2} + \ceiling{n/2} &= n\\
  \ceiling{\frac{\ceiling{x/a}}{b}} &= \ceiling{\frac{x}{ab}}\\
  \floor{\frac{\floor{x/a}}{b}} &= \floor{\frac{x}{ab}}\\
\ceiling{\frac{a}{b}} &\leq \frac{a+(b-1)}{b}\\
\floor{\frac{a}{b}} &\leq \frac{a-(b-1)}{b}\\
\end{align*}


\end{frame}

\sect{Logarithms}

\begin{align*}
  \lg^k n &= (\lg n)^k
  \\
  \lg \lg n &= \lg(\lg n)
\\
b^{\log_b(x)} &= x
\\
\log_a(xy) &= \log_a(x) + \log_a(y)
\\
\log_a(x/y) &= \log_a(x) - \log_a(y)
\\
\log_a(x^r) &= r\log_a(x)
\\
\log_a(a^r) &= r
\\
 \log_a(x) &= (\log_a(b))(\log_b(x))
\\
 \log_a(x) &= \frac{\log_b(x)}{\log_b(a)}
 \\
 \log_a(b) &= \frac{1}{\log_b(a)}
\\
a^{\log_b(c)} &= c^{\log_b(a)}
\end{align*}


\end{frame}

\sect{Factorials}
\begin{align*}
  n! &\leq n^n
  \\
  n! &= \sqrt{2\pi n}\left(\frac{n}{e}\right)^n\left(1 + \Theta\left(\frac{1}{n}\right)\right)
  \\
  &= \sqrt{2\pi n}\left(\frac{n}{e}\right)^n e^{\alpha_n}
  &\frac{1}{12n+1} < \alpha_n < \frac{1}{12n}
  \\
  n! &= o(n^n)
  \\
  n! &= \omega(2^n)
  \\
  \lg(n!) &= \Theta(n\lg n)
\end{align*}

\end{frame}

\sect{Iterated log function}

\begin{align*}
  \lg^* (n) &= \min\{i\geq 0 : \lg^{(i)} n \leq 1\}\\\\
  \lg^* 2 &= 1\\
  \lg^* 4 &= 2\\
  \lg^* 16 &= 3\\
  \lg^* 65536 &= 4\\
  \lg^* 2^{65536} &= 5\\
\end{align*}
\end{frame}


\sect{Fibonacci numbers}

\begin{align*}
  0 &= x^2 - x - 1 & \mbox{roots are $\phi$ and $\hat{\phi}$}
  \\
  \phi &= \frac{1+\sqrt{5}}{2} \approx 1.618
  \\
  \hat{\phi} &= \frac{1-\sqrt{5}}{2} \approx -0.618
  \\
  F_i &= \frac{\phi^i - \hat{\phi}^i}{\sqrt{5}}
  \\
  F_i &= \floor{\frac{\phi^i}{\sqrt{5}} + \frac{1}{2}}
\end{align*}
