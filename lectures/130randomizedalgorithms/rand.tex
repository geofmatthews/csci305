\begin{codebox}
  \Procname{$\proc{Hire-Assistant}(n)$}
  \li $\id{best}\gets 0$
  \li \For $i\gets 1$ \To $n$ \Do
  \li interview candidate $i$
  \li \If candidate $i$ is better than candidate $\id{best}$ \Do
  \li $\id{best} \gets i$
  \li hire candidate $i$
\End
\End
\end{codebox}

\vfill

\begin{codebox}
  \Procname{$\proc{Randomized-Hire-Assistant}(n)$}
  \li randomly permute the list of candidates
  \li $\id{best}\gets 0$
  \li \For $i\gets 1$ \To $n$ \Do
  \li interview candidate $i$
  \li \If candidate $i$ is better than candidate $\id{best}$ \Do
  \li $\id{best} \gets i$
  \li hire candidate $i$
\End
\End
\end{codebox}

\begin{codebox}
  \Procname{$\proc{Permute-By-Sorting}(A)$}
  \li $n \gets A.length$
  \li let $P[1...n]$ be a new array
  \li \For $i\gets 1$ \To n \Do
  \li $P[i] \gets \proc{Random}(1,n^3)$
\End
\li sort $A$ using $P$ as sort keys
\end{codebox}

\vfill

\begin{codebox}
  \Procname{$\proc{Randomize-In-Place}(A)$}
  \li $n \gets A.length$
  \li \For $i\gets 1$ \To n \Do
  \li swap $A[i]$ with $A[\proc{Random}(i,n)]$
\End
\end{codebox}
